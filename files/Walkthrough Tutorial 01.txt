admin@kali:~/SecurityClass/Tutorial-01/1.4$ cat new-sign-overflow.c 
#include "stdio.h"

int main() {
  short nb1, nb2, 
  res = 0; printf("enter first digit : "); 
  scanf("%hd", &nb1); 
  printf("enter second digit : "); 
  scanf("%hd", &nb2);

  res = nb1 + nb2;
  if ((nb1 < 0 && nb2 < 0 && res > 0) ||
      (nb1 > 0 && nb2 > 0 && res < 0)
     ) {
    printf("Overflow detected!\n");
  } else {
    printf("%hd + %hd = %hd\n", nb1, nb2, res);
  }
  return 0;
}

#include "stdlib.h"
#include "stdio.h"
#include "string.h"

#define LIMIT 4096

int main(int argc, char *argv[]){
  int i = LIMIT; 
  char * buf;
  if (argc > 2) {
    i = atoi(argv[2]); 
    if (i>LIMIT){ printf("Buffer size too large\n"); return 1;}
    if (i<1){ printf("Buffer size too small\n"); return 2;}
  }
  FILE *file;
  int nread;
  buf = malloc(i);
  file = fopen(argv[1],"r");
  if (file) {
    while ((nread = fread(buf, 1, i, file)) > 0) {
      if (nread <= i) { buf[nread] = 0; }
      printf("%s",buf);
    }
    fclose(file);
  }
  free(buf);
  return 0;
}

The fixes here are in a couple of places (I've not shown output from testing longfile). I've changed the types for all the input and usage to be the same. I've also refactored to have a single LIMIT value I use by default or update. I now check for negative values as well as being over the limit. I also fix the potential for memory allocation problems, although we probably could have a larger buffer if we wanted. 

Note this code is not perfectly safe, but it's a lot better for the problems we're looking for here. Also this code improves on bettercat.c as well.
